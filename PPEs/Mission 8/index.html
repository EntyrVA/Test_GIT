15 Exercices Mathématiques Interactifs

Exercice Actions
1. Addition de deux nombres 
function additionner() {
    const a = parseFloat(prompt("Entrez le premier nombre :"));
    const b = parseFloat(prompt("Entrez le deuxième nombre :"));
    alert(`La somme est : ${a + b}`);
}
2. Soustraction de deux nombres 
function soustraire() {

    const b = parseFloat(prompt("Entrez le deuxième nombre :"));

}
3. Multiplication de deux nombres 
function multiplier() {

    const b = parseFloat(prompt("Entrez le deuxième nombre :"));

}
4. Division de deux nombres 
function diviser() {
    const a = parseFloat(prompt("Entrez le premier nombre :"));
    const b = parseFloat(prompt("Entrez le deuxième nombre :"));
    if (b !== 0) {
        alert(`Le quotient est : ${a / b}`);
    } else {
        alert("Division par zéro impossible.");
    }
}

10. Vérifier si un nombre est premier 
function verifierNombrePremier() {
    const nombre = parseInt(prompt("Entrez un nombre :"));
    let estPremier = nombre > 1;
    for (let i = 2; i < nombre; i++) {
        if (nombre % i === 0) {
            estPremier = false;
            break;
        }
    }
    alert(`${nombre} est ${estPremier ? '' : 'pas'} un nombre premier.`);
}
11. Calcul de la somme des chiffres d'un nombre 
function sommeChiffres() {
    const nombre = prompt("Entrez un nombre :");
    const somme = nombre.split('').reduce((acc, chiffre) => acc + parseInt(chiffre), 0);
    alert(`La somme des chiffres de ${nombre} est ${somme}`);
}
12. Trouver le PGCD de deux nombres 
function trouverPGCD() {
    let a = parseInt(prompt("Entrez le premier nombre :"));
    let b = parseInt(prompt("Entrez le deuxième nombre :"));
    while (b) {
        [a, b] = [b, a % b];
    }
    alert(`Le PGCD est ${a}`);
}
13. Calcul du PPCM de deux nombres 
function trouverPPCM() {
    const a = parseInt(prompt("Entrez le premier nombre :"));
    const b = parseInt(prompt("Entrez le deuxième nombre :"));
    const ppcm = (a * b) / trouverPGCD(a, b);
    alert(`Le PPCM est ${ppcm}`);
}
14. Générer un nombre aléatoire 
function genererNombreAleatoire() {
    alert(`Nombre aléatoire : ${Math.floor(Math.random() * 100)}`);
}
15. Vérifier si un nombre est un carré parfait 
function verifierCarreParfait() {
    const nombre = parseInt(prompt("Entrez un nombre :"));
    alert(`${nombre} est ${Number.isInteger(Math.sqrt(nombre)) ? '' : 'pas'} un carré parfait.`);
}
